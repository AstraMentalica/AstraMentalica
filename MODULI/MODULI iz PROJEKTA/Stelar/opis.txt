Modul: Stelaris – Astrologija


CREATE DATABASE stelaris_astrologija;

USE stelaris_astrologija;

CREATE TABLE uporabniki (
    id INT PRIMARY KEY AUTO_INCREMENT,
    email VARCHAR(255) UNIQUE NOT NULL,
    geslo VARCHAR(255) NOT NULL,
    ime VARCHAR(100),
    datum_rojstva DATE,
    kraj_rojstva VARCHAR(100),
    stopnja_dostopa ENUM('S0','S1','S2','S3','S4','S5') DEFAULT 'S0',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE natalne_karte (
    id INT PRIMARY KEY AUTO_INCREMENT,
    uporabnik_id INT,
    datum_rojstva DATETIME NOT NULL,
    kraj_rojstva VARCHAR(100) NOT NULL,
    polozaji_planetov JSON,
    aspekti JSON,
    interpretacija TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (uporabnik_id) REFERENCES uporabniki(id)
);

CREATE TABLE horoskopi (
    id INT PRIMARY KEY AUTO_INCREMENT,
    znamenje VARCHAR(20) NOT NULL,
    datum DATE NOT NULL,
    vsebina TEXT NOT NULL,
    tip ENUM('dnevni','tedenski','mesecni') DEFAULT 'dnevni',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE clanki (
    id INT PRIMARY KEY AUTO_INCREMENT,
    naslov VARCHAR(255) NOT NULL,
    vsebina TEXT NOT NULL,
    oznake JSON,
    avtor_id INT,
    status ENUM('osnutek','objavljen','arhiviran') DEFAULT 'objavljen',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (avtor_id) REFERENCES uporabniki(id)
);
