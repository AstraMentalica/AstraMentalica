<?php
/**
 * pomocnik.php – preprost AI pomočnik za tvoj projekt
 * ============================================================
 * 1. Vnesi vprašanje o datotekah
 * 2. Skripta prebere tvoje datoteke (ali določeno mapo)
 * 3. Pošlje v DeepSeek API
 * 4. Dobiš odgovor
 * ============================================================
 */
declare(strict_types=1);

$API_KEY = getenv('DEEPSEEK_API_KEY') ?: '___VSTAVI_DEEPSEEK_API_KLJUC___'; // Nastavi v POT_SEF/.env_api (DEEPSEEK_API_KEY=...)
$ROOT = __DIR__;

// ── Obdelaj AJAX zahtevo ──────────────────────────────────────
if ($_SERVER['REQUEST_METHOD'] === 'POST' && isset($_POST['vprasanje'])) {
    header('Content-Type: application/json; charset=utf-8');
    $vprasanje = trim($_POST['vprasanje'] ?? '');
    $mapa = trim($_POST['mapa'] ?? '');
    
    if (empty($vprasanje)) {
        echo json_encode(['napaka' => 'Vpiši vprašanje.']);
        exit;
    }
    
    // Zberi datoteke iz izbrane mape (ali celoten projekt)
    $pot = $mapa ? $ROOT . '/' . $mapa : $ROOT;
    $datoteke = [];
    $iterator = new RecursiveIteratorIterator(
        new RecursiveDirectoryIterator($pot, RecursiveDirectoryIterator::SKIP_DOTS)
    );
    foreach ($iterator as $file) {
        if ($file->getExtension() === 'php' && strpos($file->getPathname(), '/AI/') === false) {
            $rel = str_replace($ROOT . '/', '', $file->getPathname());
            $vsebina = file_get_contents($file->getPathname());
            $datoteke[] = ['pot' => $rel, 'vsebina' => substr($vsebina, 0, 2000)];
        }
    }
    
    // Pošlji v DeepSeek
    $sporocilo = "Vprašanje: $vprasanje\n\n";
    $sporocilo .= "Datoteke:\n";
    foreach ($datoteke as $d) {
        $sporocilo .= "\n=== " . $d['pot'] . " ===\n" . $d['vsebina'] . "\n";
    }
    
    $sistemski = "Ti si pomočnik za PHP projekt. Odgovarjaj v slovenščini, bodisi kratek in koristen. Če ne veš, povej, da ne veš.";
    
    $ch = curl_init('https://api.deepseek.com/v1/chat/completions');
    curl_setopt_array($ch, [
        CURLOPT_RETURNTRANSFER => true,
        CURLOPT_POST => true,
        CURLOPT_POSTFIELDS => json_encode([
            'model' => 'deepseek-chat',
            'temperature' => 0.3,
            'messages' => [
                ['role' => 'system', 'content' => $sistemski],
                ['role' => 'user', 'content' => $sporocilo]
            ]
        ]),
        CURLOPT_HTTPHEADER => ['Content-Type: application/json', 'Authorization: Bearer ' . $API_KEY],
        CURLOPT_TIMEOUT => 60,
    ]);
    $r = curl_exec($ch);
    $info = curl_getinfo($ch);
    curl_close($ch);
    
    if ($info['http_code'] === 200) {
        $d = json_decode($r, true);
        $odgovor = $d['choices'][0]['message']['content'] ?? 'Brez odgovora';
        echo json_encode(['odgovor' => $odgovor]);
    } else {
        echo json_encode(['napaka' => 'HTTP ' . $info['http_code']]);
    }
    exit;
}

<script>
async function poslji() {
    const v = document.getElementById('vprasanje');
    const m = document.getElementById('mapa');
    const chat = document.getElementById('chat');
    const msg = v.value.trim();
    if (!msg) return;
    chat.innerHTML += `<div class="msg u">🧑 ${msg}</div>`;
    v.value = '';
    chat.scrollTop = chat.scrollHeight;
    try {
        const fd = new FormData();
        fd.append('vprasanje', msg);
        fd.append('mapa', m.value);
        const r = await fetch(location.href, {method:'POST', body:fd});
        const d = await r.json();
        chat.innerHTML += `<div class="msg a">🤖 ${d.odgovor || '❌ ' + d.napaka}</div>`;
    } catch(e) {
        chat.innerHTML += `<div class="msg e">❌ Napaka: ${e.message}</div>`;
    }
    chat.scrollTop = chat.scrollHeight;
}
</script>
