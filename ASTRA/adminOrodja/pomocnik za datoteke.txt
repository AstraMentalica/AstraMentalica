declare(strict_types=1);

$API_KEY = getenv('DEEPSEEK_API_KEY') ?: '___VSTAVI_DEEPSEEK_API_KLJUC___'; // Nastavi v POT_SEF/.env_api (DEEPSEEK_API_KEY=...)
$ROOT = __DIR__;
$MAPE = ['RAZVOJ', 'SISTEM', 'GLOBALNO', 'ADAPTER', 'MODULI']; // mape za pregled

// ── OBDELAJ AJAX ZAHTEVE ──────────────────────────────────────
if ($_SERVER['REQUEST_METHOD'] === 'POST' && isset($_POST['akcija'])) {
    header('Content-Type: application/json; charset=utf-8');
    $akcija = $_POST['akcija'];
    
    try {
        switch ($akcija) {
            // 1. PREGLED VSEH DATOTEK
            case 'preglej':
                $odgovor = poglej_vse();
                echo json_encode(['uspeh' => true, 'odgovor' => $odgovor]);
                break;
            
            // 2. POPRAVI VSE (KODER PREDLAGA POPRAVKE)
            case 'popravi':
                $revizija = $_POST['revizija'] ?? '';
                $odgovor = popravi_vse($revizija);
                echo json_encode(['uspeh' => true, 'odgovor' => $odgovor]);
                break;
            
            // 3. SHRANI POPRAVKE V DATOTEKE
            case 'shrani':
                $popravki = json_decode($_POST['popravki'] ?? '[]', true);
                $rezultat = shrani_popravke($popravki);
                echo json_encode(['uspeh' => true, 'rezultat' => $rezultat]);
                break;
            
            default:
                throw new Exception("Neznana akcija");
        }
    } catch (Exception $e) {
        echo json_encode(['uspeh' => false, 'napaka' => $e->getMessage()]);
    }
    exit;
}

// ── FUNKCIJE ──────────────────────────────────────────────────

function preberi_vse_datoteke(): array {
    global $ROOT, $MAPE;
    $vse = [];
    foreach ($MAPE as $mapa) {
        $pot = $ROOT . '/' . $mapa;
        if (!is_dir($pot)) continue;
        $iterator = new RecursiveIteratorIterator(
            new RecursiveDirectoryIterator($pot, RecursiveDirectoryIterator::SKIP_DOTS)
        );
        foreach ($iterator as $file) {
            if ($file->getExtension() === 'php') {
                $rel = str_replace($ROOT . '/', '', $file->getPathname());
                $vse[$rel] = file_get_contents($file->getPathname());
            }
        }
    }
    return $vse;
}

function poklici_api(string $sistem, string $sporocilo): string {
    global $API_KEY;
    $ch = curl_init('https://api.deepseek.com/v1/chat/completions');
    curl_setopt_array($ch, [
        CURLOPT_RETURNTRANSFER => true,
        CURLOPT_POST => true,
        CURLOPT_POSTFIELDS => json_encode([
            'model' => 'deepseek-chat',
            'temperature' => 0.3,
            'messages' => [
                ['role' => 'system', 'content' => $sistem],
                ['role' => 'user', 'content' => $sporocilo]
            ]
        ]),
        CURLOPT_HTTPHEADER => ['Content-Type: application/json', 'Authorization: Bearer ' . $API_KEY],
        CURLOPT_TIMEOUT => 120,
    ]);
    $r = curl_exec($ch);
    $info = curl_getinfo($ch);
    curl_close($ch);
    if ($info['http_code'] !== 200) {
        throw new Exception("HTTP napaka " . $info['http_code']);
    }
    $d = json_decode($r, true);
    return $d['choices'][0]['message']['content'] ?? 'Brez odgovora';
}

function poglej_vse(): string {
    $datoteke = preberi_vse_datoteke();
    $stevilo = count($datoteke);
    $sporocilo = "Preglej naslednjih $stevilo PHP datotek:\n\n";
    foreach ($datoteke as $pot => $vsebina) {
        $sporocilo .= "=== $pot ===\n" . substr($vsebina, 0, 1500) . "\n\n";
    }
    $sistem = "Ti si REVIZOR. Preglej datoteke in naštej vse napake (sintaksa, varnost, arhitektura). Odgovori v slovenščini.";
    return poklici_api($sistem, $sporocilo);
}

function popravi_vse(string $revizija): string {
    $sistem = "Ti si KODER. Na podlagi revizorjevih pripomb predlagaj konkretne popravke. Vrni JSON obliko: [{\"pot\":\"SISTEM/Baza.php\",\"popravek\":\"<?php ...\"}]";
    $sporocilo = "Revizor je rekel:\n" . $revizija;
    return poklici_api($sistem, $sporocilo);
}

function shrani_popravke(array $popravki): array {
    global $ROOT;
    $rezultat = [];
    foreach ($popravki as $p) {
        $pot = $ROOT . '/' . ltrim($p['pot'], '/');
        $dir = dirname($pot);
        if (!is_dir($dir)) mkdir($dir, 0755, true);
        // Varnostna kopija
        if (file_exists($pot)) {
            copy($pot, $pot . '.bak');
        }
        file_put_contents($pot, $p['popravek']);
        $rezultat[] = ['pot' => $p['pot'], 'status' => 'shranjeno'];
    }
    return $rezultat;
}

let revizija = '';
let popravki = [];

async function akcija(akcija) {
    const v = document.getElementById('vsebina');
    v.textContent = '⏳ Delam ...\n';
    
    try {
        const fd = new FormData();
        fd.append('akcija', akcija);
        
        if (akcija === 'popravi') {
            fd.append('revizija', revizija);
        }
        if (akcija === 'shrani') {
            fd.append('popravki', JSON.stringify(popravki));
        }
        
        const r = await fetch(location.href, {method: 'POST', body: fd});
        const d = await r.json();
        
        if (!d.uspeh) {
            v.textContent = '❌ ' + d.napaka;
            return;
        }
        
        switch (akcija) {
            case 'preglej':
                revizija = d.odgovor;
                v.textContent = '📋 REVIZOR:\n\n' + d.odgovor;
                document.getElementById('popravi-gumb').disabled = false;
                document.getElementById('shrani-gumb').disabled = true;
                break;
                
            case 'popravi':
                popravki = JSON.parse(d.odgovor);
                let izpis = '✏️ KODER PREDLAGA:\n\n';
                if (Array.isArray(popravki)) {
                    popravki.forEach(p => {
                        izpis += `📄 ${p.pot}\n${p.popravek}\n\n`;
                    });
                } else {
                    izpis += d.odgovor;
                }
                v.textContent = izpis;
                document.getElementById('shrani-gumb').disabled = false;
                break;
                
            case 'shrani':
                let rez = '💾 SHRANJENO:\n\n';
                if (Array.isArray(d.rezultat)) {
                    d.rezultat.forEach(r => {
                        rez += `✅ ${r.pot} → ${r.status}\n`;
                    });
                } else {
                    rez += d.rezultat;
                }
                v.textContent = rez;
                document.getElementById('shrani-gumb').disabled = true;
                break;
        }
    } catch (e) {
        v.textContent = '❌ Napaka: ' + e.message;
    }
}
</script>
